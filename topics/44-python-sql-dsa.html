SQL — Data Science Patterns
🗃️ SQL
-- Running average payment rate (window function)
SELECT customer_id, month,
  AVG(payment_ratio) OVER (PARTITION BY customer_id
    ORDER BY month ROWS BETWEEN 2 PRECEDING AND CURRENT ROW) AS rolling_3m_rate
FROM payments;

-- Cohort retention analysis
SELECT cohort_month, months_since_join,
  COUNT(DISTINCT user_id) AS retained_users,
  COUNT(DISTINCT user_id) * 1.0 / FIRST_VALUE(COUNT(DISTINCT user_id))
    OVER (PARTITION BY cohort_month ORDER BY months_since_join) AS retention_rate
FROM user_cohorts
GROUP BY cohort_month, months_since_join;

-- Customers who defaulted within 6 months of application
SELECT a.customer_id, a.app_date, d.default_date
FROM applications a
LEFT JOIN defaults d ON a.customer_id = d.customer_id
  AND d.default_date BETWEEN a.app_date AND a.app_date + INTERVAL '6 months'
WHERE d.customer_id IS NOT NULL;

    def forward(self, x):

class BinaryClassifier(nn.Module):
    def __init__(self, input_dim, hidden_dim=64):
        super().__init__()
        self.layers = nn.Sequential(
            nn.Linear(input_dim, hidden_dim),
            nn.BatchNorm1d(hidden_dim),
            nn.ReLU(),
            nn.Dropout(0.3),
            nn.Linear(hidden_dim, hidden_dim // 2),
            nn.ReLU(),
            nn.Linear(hidden_dim // 2, 1)  # raw logits
        )
    def forward(self, x):
        return self.layers(x).squeeze(-1)

TensorFlow 2.x — Core Patterns
🗃️ SQL
import tensorflow as tf
from tensorflow import keras

# ─── 1. SEQUENTIAL API (simple) ───────────────────────────────
model = keras.Sequential([
    keras.layers.Input(shape=(20,)),
    keras.layers.Dense(64, activation='relu'),
    keras.layers.BatchNormalization(),
    keras.layers.Dropout(0.3),
    keras.layers.Dense(32, activation='relu'),
    keras.layers.Dense(1, activation='sigmoid')
])

# ─── 2. FUNCTIONAL API (for complex graphs) ───────────────────
inputs = keras.Input(shape=(20,))
x = keras.layers.Dense(64, activation='relu')(inputs)
x = keras.layers.BatchNormalization()(x)
x = keras.layers.Dropout(0.3)(x)
outputs = keras.layers.Dense(1, activation='sigmoid')(x)
model = keras.Model(inputs, outputs)

# ─── 3. COMPILE + TRAIN ───────────────────────────────────────
model.compile(
    optimizer=keras.optimizers.AdamW(learning_rate=1e-3, weight_decay=1e-4),
    loss='binary_crossentropy',
    metrics=['AUC', 'Precision', 'Recall']
)

callbacks = [
    keras.callbacks.EarlyStopping(patience=5, restore_best_weights=True),
    keras.callbacks.ReduceLROnPlateau(factor=0.5, patience=3),
    keras.callbacks.ModelCheckpoint('best_model.keras', save_best_only=True)
]

history = model.fit(
    X_train, y_train,
    validation_split=0.2,
    epochs=100,
    batch_size=64,
    callbacks=callbacks
)

# ─── 4. SUBCLASSED MODEL (custom training loop) ───────────────
class CustomModel(keras.Model):
    def __init__(self):
        super().__init__()
        self.dense1 = keras.layers.Dense(64, activation='relu')
        self.out = keras.layers.Dense(1, activation='sigmoid')
    def call(self, x, training=False):
        return self.out(self.dense1(x))
    def train_step(self, data):
        X, y = data
        with tf.GradientTape() as tape:
            y_pred = self(X, training=True)
            loss = self.compute_loss(y=y, y_pred=y_pred)
        grads = tape.gradient(loss, self.trainable_variables)
        self.optimizer.apply_gradients(zip(grads, self.trainable_variables))
        return {m.name: m.result() for m in self.metrics}

# ─── 5. SAVE / EXPORT ─────────────────────────────────────────
model.save('model.keras')                   # Keras format
model.export('saved_model/')                # TF SavedModel for serving
# TFLite for mobile:
converter = tf.lite.TFLiteConverter.from_keras_model(model)
tflite_model = converter.convert()

Key Differences Deep Dive
  • Computational graph: PyTorch builds graph dynamically as you execute (eager by default). TF 2 is also eager but @tf.function compiles to static graph for speed — can be hard to debug.
  • Debugging: PyTorch = put a print() or breakpoint anywhere. TF graph mode = must use tf.print() inside the graph.
  • GPU memory: TF claims all GPU memory upfront; set tf.config.experimental.set_memory_growth(gpu, True) to avoid this.
  • DDP training: PyTorch DistributedDataParallel is cleaner API than TF's MirroredStrategy for multi-GPU.
Common Gotchas
  • PyTorch: forget zero_grad() → accumulated gradients. Forget model.eval() → dropout active during inference.
  • TF/Keras: training=True/False controls BN and Dropout — pass it in model(x, training=True). Default model.predict() sets training=False.
  • Data shapes: PyTorch expects [B, C, H, W] (NCHW); TF expects [B, H, W, C] (NHWC) — a common transposition bug.
  • Loss functions: PyTorch BCEWithLogitsLoss (logits) ≠ BCELoss (probs). Use the logits version — numerically stable.
🎯 Interview Gold

"PyTorch is my default for anything research-adjacent or using HuggingFace models — the ecosystem is now dominant. The key advantage is the dynamic computation graph: you can conditionally execute parts of the graph, insert Python debugging statements, and inspect tensors at any point. TensorFlow's advantage is deployment: TF Serving, TF Lite, and TF.js form the best end-to-end production ecosystem. Keras 3 is interesting because it abstracts over both — same code runs on PyTorch or TF backend."
